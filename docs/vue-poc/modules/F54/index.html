        function xqd:rxq-paths($state)
as map(*)* 
{

                                     let $hit:=  xqd:methods($annot?annot/.., $xqd:nsRESTXQ, $method)
                                     where $hit

                         return if(exists($hits))then $hits else map{"ALL":$annot}
                       )

        ) 
let $uris:=sort(map:keys($data))        
return $data?($uris)        
}
