function vue-api:file($url as xs:string,$protocol as xs:string)
as element(json)   
{
   let $reader:=map{
              "webfile":ufile:webfile#1,
              "xmldb":ufile:xmldb#1
              }
   let $items:=$reader($protocol=>trace("PROTO"))($url)
   return vue-api:items($items)
};
