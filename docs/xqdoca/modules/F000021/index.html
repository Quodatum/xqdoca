declare function opts:merge($opts as map(*),$defaults as map(*))
as map(*){
distinct-values ((map:keys($opts), map:keys($defaults)))
! map:entry(.,
      if(map:contains($opts,.) and map:contains($defaults,.))
      then if (map:get($opts,.) instance of map(*))

           else map:get($opts,.)
