function xqo:load-generators()
as function(*)*
{
  let $base:=resolve-uri("generators/",static-base-uri())
  for $f in file:list($base,true(),"*.xqm")
  return inspect:functions(resolve-uri($f,$base))
}

function xqo:render( $model as map(*),$opts as map(*))
as map(*)*
{
  let $funs:=xqo:load-generators()
  let $global:=(xqo:renderers($funs,$xqo:global)!xqo:render-map(.))[?name =$opts?outputs?global] 
  let $module:=(xqo:renderers($funs,$xqo:module)!xqo:render-map(.))[?name =$opts?outputs?module]
  (: add found renderers info to opts :)
  let $opts:=map:merge((map:entry(".renderers",map{"global":$global,"module":$module}),$opts))
  return (
      for $render in $global
      let $doc:= apply($render?function,[$model,$opts])
      return map{"document": $doc, 
                 "uri": $render?uri, 
                 "output":$xqo:outputs?($render?output)
               },
               
      for $render in $module, $file at $pos in $model?files
      (: override opts for destination path :)
      let $opts:=map:merge((
            map{
              "root": "../../",
              "resources": "../../resources/"
            }, $opts))
      let $doc:= apply($render?function,[$file,$opts,$model])       
      return map{"document": $doc, 
                 "uri": concat($file?href,"/",$render?uri),  
                 "output": $xqo:outputs?($render?output)
                }
              )                                      
}

function xqo:render-map( $function as function(*)?)
as map(*){
  let $ann:= inspect:function-annotations($function)
  let $key:=if(map:contains($ann,$xqo:module)) then
                $xqo:module
            else if(map:contains($ann,$xqo:global)) then
                $xqo:global
            else
               error(xs:QName("xqo:anno-map"))
   return map{
    "name": $ann?($key)[1],
     "description": $ann?($key)[2],
     "function": $function,
     "type": $key,
     "uri": $ann?($xqo:ann-output)[1],
     "output": $ann?($xqo:ann-output)[2]
} 
}
