function dba:db-drop(
  $names  as xs:string*
) as empty-sequence() {
  try {
    $names ! db:drop(.),
    util:redirect($dba:CAT, map { 'info': util:info($names, 'database', 'dropped') })
  } catch * {
    util:redirect($dba:CAT, map { 'error': $err:description })
  }
}

declare
  %updating
  %rest:GET
  %rest:path("/dba/db-drop")
  %rest:query-param("name", "{$names}")
function dba:db-drop(
  $names  as xs:string*
) as empty-sequence() {
  try {
    $names ! db:drop(.),
    util:redirect($dba:CAT, map { 'info': util:info($names, 'database', 'dropped') })
  } catch * {
    util:redirect($dba:CAT, map { 'error': $err:description })
  }
};
