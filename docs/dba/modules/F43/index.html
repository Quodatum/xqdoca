declare
  %rest:POST
  %rest:path("/dba/settings")
function dba:settings-save(
) as element(rest:response) {
  options:save(map:merge(Request:parameter-names() ! map:entry(., Request:parameter(.)))),
  web:redirect($dba:CAT, map { 'info': 'Settings were saved.' })
};
